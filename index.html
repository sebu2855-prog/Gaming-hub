let videos = JSON.parse(localStorage.getItem("videos")) || []; function uploadVideo() { let title = document.getElementById("title").value; let fileInput = document.getElementById("file"); if (title === "") { alert("Title likho!"); return; } if (fileInput.files.length === 0) { alert("Video select karo!"); return; } let file = fileInput.files[0]; let url = URL.createObjectURL(file); videos.push({ title: title, url: url }); localStorage.setItem("videos", JSON.stringify(videos)); showVideos(); document.getElementById("title").value = ""; fileInput.value = ""; } function showVideos() { let list = document.getElementById("videoList"); list.innerHTML = ""; document.getElementById("count").innerText = videos.length; videos.forEach((video, index) => { let li = document.createElement("li"); li.innerHTML = ` ${video.title} `; li.onclick = function() { document.getElementById("player").src = video.url; }; list.appendChild(li); }); } function deleteVideo(index) { videos.splice(index, 1); localStorage.setItem("videos", JSON.stringify(videos)); showVideos(); } // page load pe auto show showVideos();
